0.1 table province:
_ province_id : INT (PK , AI)
_ province_name: varchar(50)
0.2 table district:
_ district_id: INT (PK , AI)
_ district_name: varchar(50)
_ province_id: INT (FK , from table province)

0.3 table ward:
_ ward_id : INT (PK , AI)
_ ward_name : varchar(50)
_ district_id : INT (FK , from table district)

2.table `account`
- account_id : INT (PK, AI)
- account_name: VARCHAR(50)
- account_password: VARCHAR(255)
- account_email : VARCHAR(50)
- account_enable : BIT(1) (0 : false; 1 : true) --> dùng để remove 
- account_logout_time : DATETIME
- user_id : INT (FK from table `user`)


3.table `role`
- role_id : INT (PK, AI)
- role_name : VARCHAR(20) (value : 'ADMIN', 'MEMBER')



4.table `account_role`
- account_role_id : INT(PK, AI)
- account_id : INT (FK from table `account`)
- role_id : INT (FK from table `role`)

5.5 table type_product
_ type_product_id : INT (PK , AI)
_ type_product_name: varchar(50)

5.table `product`
- product_id : INT (PK, AI)
- product_name : VARCHAR(50)
- product_price : DOUBLE
- product_type_id: INT (FK , from table type_product)
- product_price_step : DOUBLE
- product_service_fee : DOUBLE (default : NULL, set sau khi có last_price)
- product_quantity : INT (default : 1)
- product_last_price : DOUBLE (default : NULL, ko tính service_fee)
- product_description : TEXT
- product_status_id : INT (FK from table `product_status`)
- product_register_time : DATETIME (ngày up product)
- product_auction_time : INT(user tự nhập, đơn vị h, max = 24h) 
- product_end_time :DATETIME(default : NULL, value = set sau khi thanh toán thành công)
- account_id : INT (FK from table `account`)


6.table `product_image`
- product_image_id : INT (PK, AI)
- image : TEXT
- product_id : INT (FK from table `product`)



7.table `product_status`
- product_status_id : INT (PK, AI)
- status_name : VARCHAR(50) (value : 1.'pending'-'đang chờ duyệt', --> là default sau khi up
                    2.'approved'-'đã duyệt/đang đấu giá',--> sau khi admin duyệt
                    3.'purchasing'-'đang chờ thanh toán'
                    4.'completed'-'đã thanh toán'
                    5.'failed'-'thanh toán thất bại'
                    6.'closed'-'khóa sản phẩm/ không đồng ý duyệt'

8.table `auction`
_ auction_id : INT (PK , AI)
_ account_id: INT (FK from table account)
_ product_id: INT (FK from table product)
_ price : Double.
_ time_auction: DateTime.
_ status : varchar(50).


9.table `product_transaction`
_ product_transaction_id : INT (PK , AI)
_ product_id : INT (FK , from table product)
_ account_id: INT (FK , from table account)
_ status : varchar(20) (value : pending , success , fail)

9.5 table order_products
_ order_products_id : INT (PK , AI)
_ product_id : INT (FK , from table product)
_ order_id : INT (FK ,from table product_transaction)

10.table `order`
_ order_id: INT (PK , AI)
_ account_id: INT (FK , from table account)
_ ward_id: INT (FK , from table ward)
_ address: varchar(50)
_ phone: varchar(20)
_ guide: Text
_ total: Double
_ method_pay: bit (0: false => ko sử dụng paypal , 1: true => sử dụng paypal)

11.table comment
_ comment_id: INT (PK , AI)
_ content: TEXT
_ comment_time: Datetime
_ account_id: INT (FK, from table account)
_ product_id: INT (FK , from table product)
